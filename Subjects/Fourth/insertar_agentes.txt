  [\vspace{1ex}\titlerule]

  {\Large\bfseries}
  {\thesection}{1em}{}

  {\large\bfseries}
  {\thesubsection}{1em}{}

\titleformat{\subsubsection}
  {\normalsize\bfseries\itshape}
  {\thesubsubsection}{1em}{}
